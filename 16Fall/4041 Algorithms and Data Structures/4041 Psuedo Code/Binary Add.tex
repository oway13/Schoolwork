Binary-Add(A, B)				
	n = A.length					1
	carry = 0					1
	C =  new array, length n+1	1
	for i = 0 to n-1				n
		sum = A[i]+B[i]+carry	n-1
		if sum == 3				n-1
			carry = 1			n-1
			C[i] = 1				n-1
		else if sum == 2			n-1
			carry = 1			n-1
			C[i] = 0				n-1
		else						n-1
			carry = 0			n-1
			C[i] = sum			n-1
	C[n] = carry					1
